include(BuildQtModule)

if (WIN32)
    set(ADD_DEFINITIONS UTCOMPONENT_DLL) #UTComponent_DLL UTCOMPONENT_LIB
else()
    set(ADD_DEFINITIONS)
endif()

set(MODULE_MODULE_NAME UTComponent)
set(MODULE_IDE_FOLDER UI)

set(MODULE_HEADER_BASE_DIR ${CMAKE_CURRENT_SOURCE_DIR})

set(MODULE_HEADER_FILES #for include and install header
    include/UTComponentExport.h
    UTMenu/MenuItemModel.h
)
set(MODULE_SOURCE_PRIVATE
    include/LoggerDefine.h
    UTMenu/MenuItemModel.cpp
)
set(MODULE_DEPENDENCIES_PUBLIC
)
set(MODULE_DEPENDENCIES_PRIVATE
    Qt6::Gui Qt6::Quick
    MasterLog 
    UIDataStruct
    UIComponentBase
)

set(MODULE_BUILD_INTERFACE_DIR ${CMAKE_CURRENT_SOURCE_DIR}/include)
set(MODULE_INSTALL_INTERFACE_DIR include)
set(MODULE_BUILD_PRIVATE_DIR src)

set(MODULE_QML_TARGET_URI
    ${MODULE_MODULE_NAME}
)
set(MODULE_QML_TARGET_FILES
    UTWindow/UTDialog.qml
    UTWindow/UTWindow.qml
    UTWindow/UTTest.qml
    UTMenu/UTMenu.qml
    UTMenu/UTMenuItem.qml
    UTMenu/CreateMenus.js
    UTButton/UTButton.qml
    UTFocusItem/UTFocusItem.qml
    UTComponentUtil/UTComponentUtil.js
)
set(MODULE_QML_PUBLIC_BUILD_INTERFACE_FOLDER
    ${CMAKE_CURRENT_SOURCE_DIR}/UTMenu
)

BuildQtModule(
MODULE_NAME ${MODULE_MODULE_NAME}
TARGET_SOURCE_PRIVATE ${MODULE_SOURCE_PRIVATE}
TARGET_SOURCE_HEADER_BASE_DIR ${MODULE_HEADER_BASE_DIR}
TARGET_SOURCE_PUBLIC_HEADER ${MODULE_HEADER_FILES}
TARGET_DEPENDENCIES_PRIVATE ${MODULE_DEPENDENCIES_PRIVATE}
TARGET_DEPENDENCIES_PUBLIC ${MODULE_DEPENDENCIES_PUBLIC}
TARGET_INCLUDE_DIRECTORIES_BUILD_INTERFACE  ${MODULE_BUILD_INTERFACE_DIR}
TARGET_INCLUDE_DIRECTORIES_INSTALL_INTERFACE ${MODULE_INSTALL_INTERFACE_DIR}
TARGET_INCLUDE_DIRECTORIES_PRIVATE ${MODULE_BUILD_PRIVATE_DIR}
TARGET_DEFINITIONS ${ADD_DEFINITIONS}
IDE_FOLDER ${MODULE_IDE_FOLDER}
QML_TARGET_URI ${MODULE_QML_TARGET_URI}
QML_TARGET_FILES ${MODULE_QML_TARGET_FILES}
QML_PUBLIC_BUILD_INTERFACE_FOLDER ${MODULE_QML_PUBLIC_BUILD_INTERFACE_FOLDER}
)
