include(TemplateTools)
# 设置资源字符串 JSON 文件路径
set(RESOURCE_IDE_FOLDER commonhead/codegen)
set(RESOURCE_OUT_PUT_DIR ${UCF_CODE_GENERATOR_DIR}/UIResourceAsset)

set(TOKEN_OUT_PUT_FILE UIAssetToken.h)
set(TOKEN_OUTPUT_PATH ${RESOURCE_OUT_PUT_DIR}/${TOKEN_OUT_PUT_FILE})
set(TOKEN_TEMEPLATE_FILE ${CMAKE_CURRENT_SOURCE_DIR}/resource/templates/UIAssetToken.h.j2)

generate_from_template(
    TEMPLATE_FILE ${TOKEN_TEMEPLATE_FILE}
    INPUT_FILE ${GLOBAL_ASSET_TOKEN_JSON_PATH}
    OUTPUT_FILE ${TOKEN_OUTPUT_PATH}
    GFT_TOKEN_TARGET
)

set(CONVERTER_OUT_PUT_FILE UIAssetConvert.cpp)
set(CONVERTER_OUTPUT_PATH ${RESOURCE_OUT_PUT_DIR}/${CONVERTER_OUT_PUT_FILE})
set(CONVERTER_TEMEPLATE_FILE ${CMAKE_CURRENT_SOURCE_DIR}/resource/templates/UIAssetConvert.cpp.j2)

generate_from_template(
    TEMPLATE_FILE ${CONVERTER_TEMEPLATE_FILE}
    INPUT_FILE ${GLOBAL_ASSET_TOKEN_JSON_PATH}
    OUTPUT_FILE ${CONVERTER_OUTPUT_PATH}
    GFT_CONVERT_TARGET
)

include(BuildQtModule)

if (WIN32)
    set(ADD_DEFINITIONS UIResourceAssetLoader_DLL)
else()
    set(ADD_DEFINITIONS)
endif()

set(MODULE_MODULE_NAME UIResourceAssetLoader)
set(MODULE_IDE_FOLDER UI/ResourceLoader)

set(MODULE_HEADER_BASE_DIR ${CMAKE_CURRENT_SOURCE_DIR}/include ${RESOURCE_OUT_PUT_DIR})

set(MODULE_HEADER_FILES #for include and install header
    include/UIResourceAssetLoader/UIResourceAssetLoaderExport.h
    include/UIResourceAssetLoader/UIResourceAssetLoader.h
    ${TOKEN_OUTPUT_PATH}
)
set(MODULE_SOURCE_PRIVATE
    src/UIResourceAssetLoader.cpp
    src/LoggerDefine.h
    ${CONVERTER_OUTPUT_PATH}
)
set(MODULE_DEPENDENCIES_PUBLIC
    Qt6::Qml Qt6::Core
)
set(MODULE_DEPENDENCIES_PRIVATE
    ResourceAssetLoader
    LoggerUtils
)

set(MODULE_BUILD_INTERFACE_DIR ${MODULE_HEADER_BASE_DIR} ${RESOURCE_OUT_PUT_DIR})
set(MODULE_INSTALL_INTERFACE_DIR include)
set(MODULE_BUILD_PRIVATE_DIR src)

BuildQtModule(
MODULE_NAME ${MODULE_MODULE_NAME}
TARGET_SOURCE_PRIVATE ${MODULE_SOURCE_PRIVATE}
TARGET_SOURCE_HEADER_BASE_DIR ${MODULE_HEADER_BASE_DIR}
TARGET_SOURCE_PUBLIC_HEADER ${MODULE_HEADER_FILES}
TARGET_ADD_LINK_LIBRARY_PRIVATE ${MODULE_DEPENDENCIES_PRIVATE}
TARGET_ADD_LINK_LIBRARY_PUBLIC ${MODULE_DEPENDENCIES_PUBLIC}
TARGET_ADD_DEPENDENCIES ${GFT_TOKEN_TARGET} ${GFT_CONVERT_TARGET}
TARGET_INCLUDE_DIRECTORIES_BUILD_INTERFACE  ${MODULE_BUILD_INTERFACE_DIR}
TARGET_INCLUDE_DIRECTORIES_INSTALL_INTERFACE ${MODULE_INSTALL_INTERFACE_DIR}
TARGET_INCLUDE_DIRECTORIES_PRIVATE ${MODULE_BUILD_PRIVATE_DIR}
TARGET_DEFINITIONS ${ADD_DEFINITIONS}
IDE_FOLDER ${MODULE_IDE_FOLDER}
)
